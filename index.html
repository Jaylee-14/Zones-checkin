Set up Google Sheets

This takes about 3 minutes. You only do it once. Every check-in will then appear in your spreadsheet automatically.

1
Go to sheets.google.com and create a new spreadsheet. Name it "4C Zones Check-In Log".
2
In the spreadsheet, click Extensions → Apps Script.
3
Delete everything in the editor and paste in this code:
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Log") || ss.insertSheet("Log"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Timestamp", "Session", "Student", "Zone", "Feeling"]); sheet.getRange(1,1,1,5).setFontWeight("bold"); } var timestamp = new Date().toLocaleString("en-AU"); var rows = data.rows; for (var i = 0; i < rows.length; i++) { sheet.appendRow([ timestamp, rows[i].session, rows[i].student, rows[i].zone, rows[i].feeling ]); } return ContentService .createTextOutput(JSON.stringify({status:"ok", added: rows.length})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error", message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:"ok", message:"Zones Check-In webhook active"})) .setMimeType(ContentService.MimeType.JSON); }
4
Click Save (💾), then click Deploy → New deployment.
5
Set type to Web app. Set "Execute as" to Me. Set "Who has access" to Anyone. Click Deploy.
6
Copy the Web app URL it gives you. It starts with https://script.google.com/macros/s/…
7
Paste that URL into Settings → Apps Script Webhook URL in this app. Done! ✅
